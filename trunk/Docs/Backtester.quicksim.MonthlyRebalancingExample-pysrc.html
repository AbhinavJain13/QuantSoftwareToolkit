 4  # An example which creates a monthly allocation table 
 5  # from 2004 to 2009 and uses the quickSim to produce a fund history. 
 6  # It then dumps the allocation table to a picklefile format file named allocations.pkl 
 7  # 
 8  # Drew Bratcher 
 9  #

11  #sample_alloc setup 
12  from pylab import * 
13  from qstkutil import DataAccess as da 
14  from qstkutil import timeutil as tu 
15  from qstkutil import pseries as ps 
16  from pandas import * 
17  import quickSim as simulator 
18  import matplotlib.pyplot as plt 
19  import time as t 
20  import cPickle 
21   
22  #sample_historic setup 
23  # Get first 20 S&P Symbols  
24  symbols = list(np.loadtxt('S&P500.csv',dtype='str',delimiter=',',comments='#',skiprows=0)) 
25  symbols = symbols[0:19] 
26  #Set start and end boundary times.  They must be specified in Unix Epoch 
27  tsstart = tu.ymd2epoch(2004,1,1) 
28  tsend = tu.ymd2epoch(2009,12,31) 
29   
30  # Get the data from the data store 
31  storename = "Norgate" # get data from our daily prices source 
32  fieldname = "adj_close" # adj_open, adj_close, adj_high, adj_low, close, volume 
33   
34  historic = ps.getDataMatrixFromData(storename,fieldname,symbols,tsstart,tsend) 
35  alloc_vals=.8/(len(historic.values[0,:])-1)*ones((1,len(historic.values[0,:]))) 
36  alloc=DataMatrix(index=[historic.index[0]], data=alloc_vals, columns=symbols) 
37  for date in range(0, len(historic.index)): 
38          if(historic.index[date].day==1): 
39                  alloc=alloc.append(DataMatrix(index=[historic.index[date]], data=alloc_vals, columns=symbols)) 
40  alloc[symbols[0]] = .1 
41  alloc['_CASH'] = .1 
42  output=open("allocations.pkl","wb") 
43  cPickle.dump(alloc, output) 
44   
45  funds=simulator.quickSim(alloc,historic,1000) 
46  output2=open("funds2.pkl","wb") 
47  cPickle.dump(funds, output2) 
48   
49  plt.clf() 
50  plt.plot(funds.index,funds.values) 
51  plt.ylabel('Fund Value') 
52  plt.xlabel('Date') 
53  plt.draw() 
54  savefig("funds.pdf", format='pdf')
