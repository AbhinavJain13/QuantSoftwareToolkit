10  from pylab import * 
11  from qstkutil import DataAccess as da

15  import quickSim as simulator 
16  import matplotlib.pyplot as plt 
17  import datetime as dt 
18   
19  #sample allocation table setup     
20  dates= [dt.date(2008,1,2), 
21          dt.date(2008,1,21), 
22          dt.date(2008,2,1), 
23          dt.date(2008,3,1), 
24          dt.date(2008,4,1), 
25          dt.date(2008,5,1), 
26          dt.date(2008,6,1), 
27          dt.date(2008,7,1), 
28          dt.date(2008,8,1), 
29          dt.date(2008,9,1), 
30          dt.date(2008,10,1), 
31          dt.date(2008,11,1), 
32          dt.date(2008,12,1)] 
33   
34  vals={ 
35      'XOM' :   [.2, .2, .2, .2, .2, .3, .4, .3, .2, .1, .3, .2, .1], 
36      'IBM' :   [.3, .3, .3, .3, .3, .1, .4, .3, .2, .1, .3, .2, .1], 
37      'GLD' :   [.3, .3, .3, .3, .3, .3, .2, .4, .6, .8, .4, .6, .8], 
38      '_CASH' : [.2, .2, .2, .2, .2, .3, .0, .0, .0, .0, .0, .0, .0],} 
39   
40  sample_alloc = DataMatrix(vals, index=dates)     
41   
42  #sample_historic setup 
43  # Set start and end boundary times.  They must be specified in Unix Epoch 
44  tsstart = tu.ymd2epoch(2008,1,2) 
45  tsend = tu.ymd2epoch(2008,12,31) 
46  symbols = sample_alloc.cols() 
47  symbols.pop() 
48   
49  # Get the data from the data store 
50  storename = "Norgate" # get data from our daily prices source 
51  fieldname = "adj_close" # adj_open, adj_close, adj_high, adj_low, close, volume 
52  sample_historic = ps.getDataMatrixFromData(storename,fieldname,symbols,tsstart,tsend) 
53   
54  funds=simulator.quickSim(sample_alloc,sample_historic,1000) 
55   
56  #plot funds 
57  plt.clf() 
58  plt.plot(funds.index,funds.values) 
59  plt.ylabel('Fund Value') 
60  plt.xlabel('Date') 
61  plt.draw() 
62  savefig("funds.pdf", format='pdf')
